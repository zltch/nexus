#kubernetes

kubectl create clusterrolebinding backstage-reader \
  --clusterrole=view \
  --serviceaccount=default:default

kubectl apply -f - <<EOF
apiVersion: v1
kind: Secret
metadata:
  name: backstage-sa-token
  namespace: default
  annotations:
    kubernetes.io/service-account.name: default
type: kubernetes.io/service-account-token
EOF


export K8S_MINIKUBE_TOKEN=$(kubectl -n default get secret backstage-sa-token -o json | jq -r '.data.token' | base64 --decode)

minikube addons enable metrics-server

kubectl apply -f - <<EOF
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: backstage-metrics-reader
rules:
  - apiGroups: ["metrics.k8s.io"]
    resources: ["pods", "nodes"]
    verbs: ["get", "list", "watch"]
EOF

kubectl create clusterrolebinding backstage-metrics-binding \
  --clusterrole=backstage-metrics-reader \
  --serviceaccount=default:default
